```c++
void PlayerManager::SendPacketToAllEcsCheck(HSteamNetConnection SNET, PacketHeader* Data, u32 flags) {
    for (auto& PL : m_StoredData) {
        if (!PlayerObjectExist(&PL.second) || PL.first == SteamGameManager->M_MySNETID)
            continue;
        bool PA1 = PL.second.Status.m_cbPendingUnreliable <= 256 * 1024; // 256КБ
        bool PA2 = PL.second.Status.m_cbPendingReliable <= 256 * 1024;   // 256КБ
        bool PA3 = PL.second.Status.m_flInPacketsPerSec <= 100.0f;       // 100 Packets/Sec
        bool PA4 = PL.second.Status.m_flConnectionQualityLocal >= 0.3f;  // Quality 30%
        if (PA1 && PA2 && PA3 && PA4)
            SteamOServer->SendDataTo(PL.first, Data, flags);
        else
            Msg("=[Client] Ethernet is trhrottling! cbUnreliable [%i], cbReliable [%i], Packets [%i/sec], Quality [%i]!",
                PL.second.Status.m_cbPendingUnreliable,
                PL.second.Status.m_cbPendingReliable,
                PL.second.Status.m_flInPacketsPerSec,
                PL.second.Status.m_flConnectionQualityLocal
            );
    }
}
```
